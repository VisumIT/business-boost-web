import React,{useState} from 'react';

function Products(){


    const [count, setCount] = useState(0);

    return(
        <div className="container">
            {count == 0 
                ? 
                <button type="button" className="btn btn-primary" onClick={() =>{setCount(1)}}>Primary</button> 
                : 
                <h1>1</h1> 
            }
        </div>
    )   
}

export default Products;
